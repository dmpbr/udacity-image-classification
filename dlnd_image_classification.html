def normalize(x):
    """
    Normalize a list of sample image data in the range of 0 to 1
    : x: List of image data.  The image shape is (32, 32, 3)
    : return: Numpy array of normalize data
    """
    
    #x is a numpy array of integers so we first need to create a new numpy array to hold floats
    float_array = x.astype(dtype=np.float32)
    
    #Initialize the max and min values as floats as well
    max_value = np.float32(255.0)
    min_value = np.float32(0.0)
    
    #Test for the shape of the input received
    assert x.shape != (32,32,3), "Incorrect image shape, expected (32, 32, 3) received {}".format(x.shape)
    
    #Iterate through the new float array and normalize it
    for index,value in np.ndenumerate(float_array):
        float_array[index]= (float_array[index] - min_value) / (max_value - min_value)
        
    return float_array

def conv2d_maxpool(x_tensor, conv_num_outputs, conv_ksize, conv_strides, pool_ksize, pool_strides):
    """
    Apply convolution then max pooling to x_tensor
    :param x_tensor: TensorFlow Tensor
    :param conv_num_outputs: Number of outputs for the convolutional layer
    :param conv_ksize: kernal size 2-D Tuple for the convolutional layer
    :param conv_strides: Stride 2-D Tuple for convolution
    :param pool_ksize: kernal size 2-D Tuple for pool
    :param pool_strides: Stride 2-D Tuple for pool
    : return: A tensor that represents convolution and max pooling of x_tensor
    """
    
    # TODO: Implement Function
    filter_height = conv_ksize[0]
    filter_width = conv_ksize[1]
    color_channels = x_tensor.get_shape().as_list()[3]
    output_depth = conv_num_outputs
    
    conv_strides = [1, conv_strides[0], conv_strides[1], 1]
    
    
    std_dev = 1/np.sqrt(conv_num_outputs)
    
    #Create the weights and bias. For the weights using a truncated normal which is the function
    #that helps minimizing the loss function faster as seen in the last lecture
    weights = tf.Variable(tf.truncated_normal([filter_height, filter_width, color_channels, output_depth], \
                                             mean=0.0, stddev=std_dev))
    bias = tf.Variable(tf.truncated_normal([output_depth], mean=0.0, stddev=std_dev))
    
    #Create the convolutional layer
    conv_layer = tf.nn.conv2d(x_tensor, weights, strides=conv_strides, padding='SAME')
    
    #Add the lias to the convolutional layer
    conv_layer = tf.nn.bias_add(conv_layer, bias)
    
    # Apply non-linear activation function. In this case using ReLU
    conv_layer = tf.nn.relu(conv_layer)
    
    
    #Apply Max Pooling to the convolutional layer
    #Set the max pool filter using input and with batch = 1 and channel = 1
    p_ksize = [1, pool_ksize[0], pool_ksize[1], 1]
    #Set the max pool strides using input and with batch = 1 and channel = 1
    p_strides = [1, pool_strides[0], pool_strides[1], 1]
    
    conv_layer = tf.nn.max_pool(conv_layer, ksize=p_ksize, strides=p_strides, padding='SAME')
    
    return conv_layer

def flatten(x_tensor):
    """
    Flatten x_tensor to (Batch Size, Flattened Image Size)
    : x_tensor: A tensor of size (Batch Size, ...), where ... are the image dimensions.
    : return: A tensor of size (Batch Size, Flattened Image Size).
    """
    # TODO: Implement Function
    #First thing is to get the shape of the x_tensor out as a list to be easier to manipulate:
    shape = x_tensor.get_shape().as_list()
    #Second, we define the nem dimension for the reshaped tensorflow, which, for a 4-D tensor, is given by 
    # dim = shape[1]*shape[2]*shape[3] - i.e. we ignore the batch size
    dim = np.prod(shape[1:])
    
    #Finally we return the x_tensor reshaped
    return tf.reshape(x_tensor, [-1, dim])

def fully_conn(x_tensor, num_outputs):
    """
    Apply a fully connected layer to x_tensor using weight and bias
    : x_tensor: A 2-D tensor where the first dimension is batch size.
    : num_outputs: The number of output that the new tensor should be.
    : return: A 2-D tensor where the second dimension is num_outputs.
    """
    # TODO: Implement Function
    
    std_dev = 1/np.sqrt(num_outputs)
    
    #Construct and initialize the weights with a truncated normal. Shape is dictated by the matrix multiplication
    #Input shape: [a, b] ; weights should be [b, output]
    weights = tf.Variable(tf.truncated_normal([ x_tensor.get_shape().as_list()[1],num_outputs], \
                                              mean=0.0, stddev=std_dev))
    bias= tf.Variable(tf.truncated_normal([num_outputs], mean=0.0, stddev=std_dev))
    fc = tf.add(tf.matmul(x_tensor, weights), bias)
    fc = tf.nn.relu(fc)
    
    return fc

Note on why these were the parameters chosen:¶

To decide the parameters that I used to run the final model, I iterated multiple times keeping most parameters constant and then only changing them one at a time. The details results are available in this spreadsheet: https://docs.google.com/spreadsheets/d/1h2NpQuRe-FltRkVQlxbxW0-yQIF_tcK7GtY30IPFgcA/edit#gid=0

Here are the parameters I chose and some observations:

Convolutional Layers vs Fully Connected Layers¶

1) First parameter I optimized for was number of convolutional layers vs number fully connected layers. My approach was to do all combinations possible (1 conv x 1 FC; 2 conv x 1FC; etc.) and select the one with the highest accuracy and lowest loss. After multiple iterations I observed that the following yielded the best results:

  • Number conv layers: 2 (dropout between the first 2)
  • Number FC layers: 3 (dropout between 1-2 and 2-3)

Using 3 convolutional layers didn't yield worse results per se, but it was substantially slower to run so I decided to go with 2 COV & 3 FC. Again details are in the spreadsheet

Number Outputs¶

2) AFter defining the number of layers, I played around with the num_outputs for each layer. I obtained the best results with:

  • Num outputs cov layer 1: 32 (same as input)
  • Num outputs cov layer 2: 128 (3222, being 2 the stride length)

I did try larger number of outputs but it was much slower to compute without having much better results

  • Num outputs FC 1: 1024
  • Num outputs FC 2: 512
  • Num outputs FC 3: 256
  • Num output Output Layer: 10

Stride Length¶

3) Tried multiple combinations of stride lengths and filter sizes, but realized that it worked best with height & width of the filter equal to the pooling. Stride length of 1 increased the results slightly but to the expense of a much slower run. That said, I chose:

  • Height x Width: 5 x 5
  • Stride length: 2 x 2

Both for Filter and Pool. (note: I also tried to increase the filter and decrease the pooling size - since pooling leads to some data loss by maxing several pixels - but the results weren't much better)

Dropout¶

4) Chose dropout of 0.5. 0.75 increased the results but showed all signs of overfitting so decided to go with a more aggressive dropout

Number Batches¶

5) Number of batches was very interesting. The one that yielded the best results was 256. I was actually surprised by seeing how much impact changing this to 1024 had on the accuracy (dropped from ~65% to ~42%). I guess it makes sense that a smaller batch has better results, since smaller batches result in more iterations on the weights leading to better outputs.

def test_model():
    """
    Test the saved model against the test dataset
    """

    test_features, test_labels = pickle.load(open('preprocess_test.p', mode='rb'))
    loaded_graph = tf.Graph()

    with tf.Session(graph=loaded_graph) as sess:
        # Load model
        loader = tf.train.import_meta_graph(save_model_path + '.meta')
        loader.restore(sess, save_model_path)

        # Get Tensors from loaded model
        loaded_x = loaded_graph.get_tensor_by_name('x:0')
        loaded_y = loaded_graph.get_tensor_by_name('y:0')
        loaded_keep_prob = loaded_graph.get_tensor_by_name('keep_prob:0')
        loaded_logits = loaded_graph.get_tensor_by_name('logits:0')
        loaded_acc = loaded_graph.get_tensor_by_name('accuracy:0')
        
        # Get accuracy in batches for memory limitations
        test_batch_acc_total = 0
        test_batch_count = 0
        
        for test_feature_batch, test_label_batch in helper.batch_features_labels(test_features, test_labels, batch_size):
            test_batch_acc_total += sess.run(
                loaded_acc,
                feed_dict={loaded_x: test_feature_batch, loaded_y: test_label_batch, loaded_keep_prob: 1.0})
            test_batch_count += 1

        print('Testing Accuracy: {}\n'.format(test_batch_acc_total/test_batch_count))

        # Print Random Samples
        random_test_features, random_test_labels = tuple(zip(*random.sample(list(zip(test_features, test_labels)), n_samples)))
        random_test_predictions = sess.run(
            tf.nn.top_k(tf.nn.softmax(loaded_logits), top_n_predictions),
            feed_dict={loaded_x: random_test_features, loaded_y: random_test_labels, loaded_keep_prob: 1.0})
        helper.display_image_predictions(random_test_features, random_test_labels, random_test_predictions)
